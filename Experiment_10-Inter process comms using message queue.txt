#include <stdio.h>
#include <sys/ipc.h>
#include <sys/msg.h>
#include <unistd.h>
#include <string.h>

struct msg{
    long mtype;
    char mtext[100];
};

int main(){
    int msgid;
    struct msg m;

    msgid = msgget(IPC_PRIVATE, 0666 | IPC_CREAT);

    if(fork() == 0){
        m.mtype = 1;
        strcpy(m.mtext, "Hello from Child Process");
        msgsnd(msgid, &m, sizeof(m.mtext), 0);
    }
    else{
        msgrcv(msgid, &m, sizeof(m.mtext), 1, 0);
        printf("Message received: %s\n", m.mtext);
        msgctl(msgid, IPC_RMID, NULL);
    }

    return 0;
}
