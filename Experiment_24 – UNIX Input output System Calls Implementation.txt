#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/stat.h>
#include <dirent.h>

int main() {
    int fd;
    struct stat fileStat;
    DIR *dir;
    struct dirent *entry;

    fd = open("sample.txt", O_RDWR);
    if (fd < 0) {
        perror("Open failed");
        return 1;
    }

    int flags = fcntl(fd, F_GETFL);
    printf("File Flags: %d\n", flags);

    lseek(fd, 0, SEEK_END);
    write(fd, "Appended Text\n", 14);

    stat("sample.txt", &fileStat);
    printf("File Size: %ld bytes\n", fileStat.st_size);

    close(fd);

    dir = opendir(".");
    printf("\nDirectory Listing:\n");
    while ((entry = readdir(dir)) != NULL) {
        printf("%s\n", entry->d_name);
    }
    closedir(dir);

    return 0;
}
