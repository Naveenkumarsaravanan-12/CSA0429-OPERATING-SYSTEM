#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>

sem_t s;

void* process(void* arg) {
    sem_wait(&s);
    printf("Process %ld in Critical Section\n", (long)arg);
    printf("Process %ld leaving Critical Section\n", (long)arg);
    sem_post(&s);
    return NULL;
}

int main() {
    pthread_t t1, t2;
    sem_init(&s, 0, 1);

    pthread_create(&t1, NULL, process, (void*)1);
    pthread_create(&t2, NULL, process, (void*)2);

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    return 0;
}
