#include <stdio.h>

int main() {
    int n, blocks[10], i;

    printf("Enter number of blocks: ");
    scanf("%d", &n);

    printf("Enter block numbers:\n");
    for(i=0;i<n;i++)
        scanf("%d", &blocks[i]);

    for(i=0;i<n;i++){
        if(i != n-1)
            printf("%d -> ", blocks[i]);
        else
            printf("%d -> NULL", blocks[i]);
    }
    return 0;
}
