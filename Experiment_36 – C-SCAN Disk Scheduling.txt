#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head, size, req[100], i, seek=0;

    printf("Enter number of requests: ");
    scanf("%d",&n);

    printf("Enter request sequence:\n");
    for(i=0;i<n;i++)
        scanf("%d",&req[i]);

    printf("Enter initial head position: ");
    scanf("%d",&head);

    printf("Enter disk size: ");
    scanf("%d",&size);

    for(i=0;i<n;i++){
        seek += abs(req[i] - head);
        head = req[i];
    }

    printf("Total Seek Time: %d\n", seek);
    return 0;
}
