#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>

sem_t s;

void* task(void* arg){
    sem_wait(&s);
    printf("Task Running\n");
    sem_post(&s);
    return NULL;
}

int main(){
    pthread_t t1, t2;
    sem_init(&s,0,1);

    pthread_create(&t1,NULL,task,NULL);
    pthread_create(&t2,NULL,task,NULL);

    pthread_join(t1,NULL);
    pthread_join(t2,NULL);

    return 0;
}
