#include <stdio.h>

int main(){
    int n,i,blocks[10];

    printf("Enter number of blocks: ");
    scanf("%d",&n);

    printf("Enter block numbers:\n");
    for(i=0;i<n;i++)
        scanf("%d",&blocks[i]);

    printf("Index Block contains:\n");
    for(i=0;i<n;i++)
        printf("%d ",blocks[i]);

    return 0;
}
