#include <stdio.h>

int main(){
    int n,i,blocks[10];

    printf("Enter number of blocks: ");
    scanf("%d",&n);

    printf("Enter block numbers:\n");
    for(i=0;i<n;i++)
        scanf("%d",&blocks[i]);

    printf("Linked List Representation:\n");
    for(i=0;i<n;i++){
        if(i<n-1)
            printf("%d -> ",blocks[i]);
        else
            printf("%d -> NULL",blocks[i]);
    }
    return 0;
}
