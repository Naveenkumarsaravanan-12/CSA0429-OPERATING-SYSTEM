#include <stdio.h>
#include <sys/stat.h>

int main(){
    struct stat fileStat;

    if(stat("sample.txt",&fileStat)<0){
        perror("stat");
        return 1;
    }

    printf("Owner ID: %d\n", fileStat.st_uid);
    printf("Group ID: %d\n", fileStat.st_gid);

    printf("Permissions: %o\n", fileStat.st_mode & 0777);

    return 0;
}
