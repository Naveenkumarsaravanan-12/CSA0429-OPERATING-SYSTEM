#include <stdio.h>

int main() {
    int n, i, j, t;
    int bt[10], pr[10], wt[10], tat[10];
    float awt = 0, atat = 0;

    printf("Enter n: ");
    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        scanf("%d %d", &bt[i], &pr[i]);   // burst time, priority
    }

    // Sort by priority (lower value = higher priority)
    for (i = 0; i < n; i++)
        for (j = i + 1; j < n; j++)
            if (pr[i] > pr[j]) {
                t = pr[i]; pr[i] = pr[j]; pr[j] = t;
                t = bt[i]; bt[i] = bt[j]; bt[j] = t;
            }

    wt[0] = 0;
    for (i = 1; i < n; i++)
        wt[i] = wt[i - 1] + bt[i - 1];

    for (i = 0; i < n; i++) {
        tat[i] = wt[i] + bt[i];
        awt += wt[i];
        atat += tat[i];
        printf("P%d WT=%d TAT=%d\n", i + 1, wt[i], tat[i]);
    }

    printf("Avg WT=%.2f  Avg TAT=%.2f\n", awt / n, atat / n);
    return 0;
}
